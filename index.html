Problem 1
Write a program to print the points inside a circle at desire quadrant

Solution 

            
#include 
#include 
int main()
{
    int q, x = 0, y = 0, r;
    printf("\nEnter the radius: ");
    scanf("%d", &r);
    printf("\nEnter the Quadrant number: ");
    scanf("%d", &q);

    switch (q)
    {
    case 1:
        while (x <= r)
        {
            while (y <= r)
            {
                if ((x * x) + (y * y) <= (r * r)){
                    printf("\n(%d,%d)", x, y);
                }
                
                y++;
            }
            x++;
            y = 0;
        }
        break;
    case 2:
        while (x >= -r)
        {
            while (y <= r)
            {
                if ((x * x) + (y * y) <= (r * r)){
                    printf("\n(%d,%d)", x, y);
                }
                y++;
            }
            x--;
            y = 0;
        }
        break;
    case 3:
        while (x >= -r)
        {
            while (y >= -r)
            {
                if ((x * x) + (y * y) <= (r * r)){
                    printf("\n(%d,%d)", x, y);
                }
                y--;
            }
            x--;
            y = 0;
        }
        break;
    case 4:
        while (x <= r)
        {
            while (y >= -r)
            {
                if ((x * x) + (y * y) <= (r * r)){
                    printf("\n(%d,%d)", x, y);
                }
                y--;
            }
            x++;
            y = 0;
        }
        break;

    default:
        break;
    }

    return 0;
}
